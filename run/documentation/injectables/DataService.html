import { Injectable } from "@angular/core";
import { Response, RequestOptions } from "@angular/http";

import { Observable } from "rxjs/Observable";
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from "rxjs/operator/map";
import { Task } from "../models/task";
import { Project } from "../models/projects";
import { NotificationsService } from "../services/notifications.service";


/**
* Сервис для работы с данными.
*/
@Injectable()
export class DataService {

    private url = "http://localhost:2403";

    currentProject;

    constructor(private http: HttpClient, private _notifService: NotificationsService) { }

    // Отправка GET запроса нв сервер

    /**
    * Метод получения списка проектов.
    */
    public getProjects() {
        return this.http.get(this.url + '/projects')
            .map(function (res) {
                let projects: Project[] = [];
                for (let i in res) projects.push(res[i]);
                return projects;
            }) // преобразовывает ответ в массив экземпляров Product.
            .catch(this.handleError);
    }

    /**
    * Метод получения проекта по идентификатору.
    */
    getProject(id) {
        let project = this.http.get(this.url + '/projects/' + id)
            .map(function (res) {
                let project = {};

                for (let key in res) {
                    project[key] = res[key];
                }
                return project;
            });
        return project;
    }

    /**
    * Обновление проекта.
    */
    updateProject(project) {
        // console.log(project);
        if (project.id) {
            return this.http.put(this.url + '/projects/' + project.id, project).catch(this.handleError).map(
                res => {
                    return res
                },
                err => {
                    this._notifService.setNotification('Ошибка обновления проекта', 1);
                    return err;
                }
            )
        }
    }

    /**
    * Создание нового проекта.
    */
    createProject(project) {
        return this.http.post(this.url + '/projects', project).map(
            res => {
                this._notifService.setNotification('Проект «' + project.name + '» успешно создан', 1);
                return res
            },
            err => {
                this._notifService.setNotification('Ошибка создания задачи', 2);
                return err
            }
        );
    }

    /**
    * Удаление проекта.
    */

    deleteProject(project) {
        return this.http.delete(this.url + '/projects/' + project.id).map(
            res => {
                this._notifService.setNotification('Проект «' + project.name + '» успешно удален', 1);
                return res;
            },
            err => {
                this._notifService.setNotification('Ошибка удаления проекта', 2);
                return err
            }
        );
    }



    // TASKS

    /**
    * Получение всех задач.
    */
    getTasks(projectId?, user?) {
        let tasks = this.http.get(this.url + '/tasks')
            .map(function (res) {
                let tasks: Task[] = [];

                if (projectId) { // Загрузка задача для конкретного проекта
                    for (let i in res) {
                        if (res[i].projectId === projectId) {
                            tasks.push(res[i])
                        }
                    }

                }

                // Если открыли пункт "Мои задачи"
                if (user && !projectId) {
                    for (let i in res) {
                        tasks.push(res[i]);
                    }
                }

                else if (!user && !projectId) {
                    for (let i in res) {
                        tasks.push(res[i]);
                    }
                }


                return tasks;
            }).catch(this.handleError);

        //console.log(tasks);
        return tasks;
    }

    /**
    * Получение задачи.
    */
    getTask(id) {
        let task = this.http.get(this.url + '/tasks/' + id).map(
            (res) => {
                let task = new Task();

                for (let key in res) {
                    task[key] = res[key];
                }
                return task;
            }
        )

        return task;
    }


    /**
    * Обновление задачи.
    */

    updateTask(data) {
        return this.http.put(this.url + '/tasks/' + data.id, data).map(
            res => this._notifService.setNotification('Задача «' + data.name + '» успешно обновлена', 1),
            err => this._notifService.setNotification('Ошибка обновления задачи', 2)
        )
    }

    /**
    * Создание задачи.
    */
    createTask(data) {
        return this.http.post(this.url + '/tasks', data).map(
            res => {
                this._notifService.setNotification('Задача «' + data.name + '» успешно создана', 1);
                return res;
            },
            err => {
                this._notifService.setNotification('Ошибка добавления задачи', 2);
                return err
            }
        ).catch(this.handleError);
    }

    /**
    * Удаление задачи по идентификатору.
    */
    deleteTask(task) {
        return this.http.delete(this.url + '/tasks/' + task.id).map(
            res => this._notifService.setNotification('Задача «' + task.name + '» успешно удалена', 1),
            res => this._notifService.setNotification('Ошибка удаления задачи', 2),
        );
    }

    // COMMENTS

    /**
    * Получение всех комментариев.
    */
    getComments(taskId) {
        let comments = this.http.get(this.url + '/comments').map(
            (res) => {
                let comments = [];

                for (let i in res) {
                    if (res[i].taskId == taskId) {
                        let comment = {}
                        for (let key in res[i]) {
                            comment[key] = res[i][key];
                        }

                        comments.push(comment)
                    }
                }
                return comments
            }
        );

        return comments;
    }

    /**
    * Отправить комментарий.
    */
    sendComment(comment) {
        return this.http.post(this.url + '/comments', comment).map(
            res => {
                this._notifService.setNotification('Комментарий успешно добавлен', 1);
                return res
            },
            err => this._notifService.setNotification('Ошибка добавления комментария', 2)
        )
    }

    /**
    * Удалить комментарий
    */
    deleteComment(id) {
        return this.http.delete(this.url + '/comments/' + id)
    }


    // USERS

    /**
    * Получение массива всех пользователей.
    */

    getUsers() {
        return this.http.get(this.url + '/users').map(
            (res) => {
                let users = [];

                for (let i in res) users.push(res[i])

                return users;
            }
        )
    }

    logIn(data) {
        return this.http.post(this.url + '/users/login', data, ).catch(this.handleError);
    }


    /**
    * Отображение данных пользователя по идентификатору.
    */

    getUser(id): Observable<any> {
        return this.http.get(this.url + '/users/' + id);
    }


    /**
    * Обновление пользователя по идентификатору.
    */

    updateUser(user) {
        return this.http.put(this.url + '/users/' + user.id, user);
    }

    createUser(user) {
        return this.http.post(this.url + '/users/', user).map(
            res => {
                this._notifService.setNotification('Новый сотрудник успешно добавлен', 1);
                return res;
            },
            err => this._notifService.setNotification('Ошибка добавления сотрудника', 2)
        )
    }

    deleteUser(user) {
        return this.http.delete(this.url + '/users/' + user.id).map(
            res => this._notifService.setNotification('Сотрудник успешно удален', 1),
            err => this._notifService.setNotification('Ошибка удаления сотрудника', 2)
        )
    }


    logout() {
        sessionStorage.clear();
    }


    /**
    * Получить список прикрепленных проектов.
    */
    getFixedProjects(currentUser) {
        return this.http.get(this.url + '/user-fixed-projects').map(
            res => {
                let projects = []
                for (let i in res) {
                    if (res[i]['userId'] == currentUser.id) {
                        projects.push(res[i]);
                    }

                }
                return projects;
            }
        )
    }

    /**
    * Прикрепить проект в избранное.
    */
    bindProject(project, user) {
        this.getFixedProjects(user).subscribe(
            res => {
                let exist: boolean;

                console.log(res);
                for (let i in res) {
                    console.log(res);
                    if (res[i]['projectId'] == project['id']) {
                        console.log('Задача уже закреплена, надо открепить');
                        this.http.delete(this.url + '/user-fixed-projects/' + res[i]['id']).subscribe(
                            (res) => { console.log(res), exist = true },
                            err => console.log(err)
                        )
                        return;
                    }
                }

                if (!exist) this.http.post(this.url + '/user-fixed-projects', {
                    userId: user.id,
                    projectId: project.id,
                    projectName: project.name
                }).subscribe()



            }
        )
    }



    getEvents() {
        return this.http.get(this.url + '/events')
    }

    sendEvent(user, projectName?, taskName?, action?) {
        return this.http.post(this.url + '/events', {
            user: user,
            projectName: projectName,
            taskName: taskName,
            action: action
        })
    }


    handleError(error: any, cought: Observable<any>): any {
        let message = "";

        if (error instanceof Response) {
            let errorData = error.json().error || JSON.stringify(error.json());
            message = `${error.status} - ${error.statusText || ''} ${errorData}`
        } else {
            message = error.message ? error.message : error.toString();
        }

        //console.error(message);

        return Observable.throw(message);
    }

}
