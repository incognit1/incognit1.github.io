import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { DataService } from '../SHARED/services/data.service';
import { Input } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { AuthService } from '../SHARED/services/auth.service';

/**
* Компонент комментариев.
* Используется только внутри компонента задачи.
*/
@Component({
  selector: 'app-comments',
  templateUrl: './comments.component.html',
  styleUrls: ['./comments.component.css'],
})

export class CommentsComponent implements OnInit {
  comments = [];
  commentsForm;
  @ViewChild('commentElem') commentElem: ElementRef;

  @Input()
  task: string;

  constructor(private dataService: DataService, private fb: FormBuilder, private authService: AuthService) { }

  ngOnInit() {
    this.getComments();
    console.log( this.task["name"]);
    this.commentsForm = this.fb.group({
      userId: this.authService.currentUser["id"],
      taskId: this.task["id"],
      taskName: this.task["name"],
      body: '',
      author: this.authService.currentUser["lastName"] + " " + this.authService.currentUser["firstName"],
      date: new Date(),
      imgLink: this.authService.currentUser["imgLink"]
    });


  }

  getComments() {
    this.dataService.getComments(this.task["id"]).subscribe(
      (res) => {
        this.comments = res;
      }
    )
  }

  sendComment() {
    this.dataService.sendComment(this.commentsForm.value).subscribe(
      res => {
        this.dataService.sendEvent(this.commentsForm.value.author, undefined, this.commentsForm.value.taskName, 'добавил комментарий к задаче').subscribe();

        this.dataService.getProject(this.task["projectId"]).subscribe(
          res => {
            let notifUserId = '';

            if (this.authService.currentUser.id == res["managerId"]) this.pushNotification(this.task["workerId"]);

            else if (this.authService.currentUser.id == this.task["workerId"]) this.pushNotification(res["managerId"])

            else {
              this.pushNotification(res["managerId"]);
              this.pushNotification(this.task["workerId"]);
            }

          }
        )

        this.commentElem.nativeElement.value = "";
        this.getComments();
      }
    )
  }

  deleteComment(id) {
    this.dataService.deleteComment(id).subscribe(
      (res) => {
        console.log(res);
        this.dataService.sendEvent(this.commentsForm.value.author, undefined, this.commentsForm.value.taskName, 'удалил комментарий к задаче').subscribe();
        this.getComments()
      }
    );
  }

  pushNotification(notifUserId) {
    this.dataService.getUser(notifUserId).subscribe(
      (result) => {

        let user = result;
        let notification = {
          author: this.commentsForm.value.author,
          action: 'добавил комментарий к задаче',
          object: {
            name: this.commentsForm.value.taskName,
            type: 2,
            id: this.task["id"]
          },
          wathed: false,
          button: false
        }

        if (!user.notifications) user.notifications = [];

        user.notifications.push(notification);

        this.dataService.updateUser(user).subscribe();
      }
    )
  }


}
