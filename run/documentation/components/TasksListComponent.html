import { Component, OnInit } from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import { DataService } from '../SHARED/services/data.service';
import { NgbDropdownConfig, NgbTabsetConfig } from '@ng-bootstrap/ng-bootstrap';
import { AuthService } from '../SHARED/services/auth.service';
import { FilterData } from '../SHARED/pipes/filter-data.pipe';
import { Observable } from 'rxjs';
import { debounceTime, map } from 'rxjs/operators';
import { AppearanceTrigger } from '../SHARED/animations';
import { PagerService } from '../SHARED/services/pager.service';

/**
* Компонент отображения всех задач.
*/
@Component({
  selector: 'app-tasks-list',
  templateUrl: './tasks-list.component.html',
  styleUrls: ['./tasks-list.component.css'],
  animations: [AppearanceTrigger]
})
export class TasksListComponent implements OnInit {
  tasks;
  msg: string;
  titleProject: string;
  projectId: string;
  myToDoList = false;



  dataOperations = {
    filterType: "1",
    sortType: "1",
    sortOrder: true
  }

  constructor(private router: Router, private dataService: DataService,
    private activatedRoute: ActivatedRoute, config: NgbDropdownConfig, configTabs: NgbTabsetConfig, private authService: AuthService, private pagerService: PagerService) {
    config.placement = 'bottom-right';
    configTabs.justify = 'center';
    configTabs.type = 'pills';
  }

  ngOnInit() {
    this.activatedRoute.params.forEach((params) => {
      this.getTasks(params["id"], this.authService.currentUser);
      this.projectId = params["id"];
      this.msg = "";
      this.myToDoList = !params["id"];
    }
    );

    this.titleProject = this.activatedRoute.snapshot.queryParams["project"]; // project name
  }

  getTasks(id, worker?) {

    this.dataService.getTasks(id, worker).subscribe(
      res => {
        this.tasks = res;
        if (!res.length) this.msg = 'Задачи отсутствуют! Нажмите на кнопку ниже чтобы добавить первую задачу.';
      }
    );

  }


  openTask(id) {
    this.router.navigate([id], { relativeTo: this.activatedRoute });
  }

  createTask() {
    this.router.navigate(['new-task'], { relativeTo: this.activatedRoute });
  }

  deleteTask(task) {
    this.dataService.deleteTask(task).subscribe(
      (res) => {
        this.dataService.getProject(task.projectId).subscribe(
          re => {
            let p = re;
            p['taskQuantity']--;

            this.dataService.updateProject(p).subscribe(
              res => {
                this.getTasks(this.projectId, this.authService.currentUser.id);
              }
            )

          }
        )
      }
    )
  }

  sortFunc(type) {
    this.dataOperations["sortOrder"] = !this.dataOperations["sortOrder"];
    this.dataOperations["sortType"] = type;
  }


  search = (text$: Observable<string>) =>
    text$.pipe(
      debounceTime(200),
      map(term => term === '' ? []
        : this.tasks.filter(v => v.name.toLowerCase().indexOf(term.toLowerCase()) > -1).slice(0, 10))
    );

  formatter = (x: { name: string }) => x.name;

}
