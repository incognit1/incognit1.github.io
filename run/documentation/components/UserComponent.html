import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { DataService } from '../SHARED/services/data.service';
import { Router, ActivatedRoute } from '@angular/router';
import { FormBuilder, Validators } from '@angular/forms';
import { AppearanceTrigger } from '../SHARED/animations';
import { AuthService } from '../SHARED/services/auth.service';
import { NotificationsService } from '../SHARED/services/notifications.service';

/**
* Компонент формы текущего пользователя.
*/
@Component({
  selector: 'app-user',
  templateUrl: './user.component.html',
  styleUrls: ['./user.component.css'],
  animations: [AppearanceTrigger]
})
export class UserComponent implements OnInit {
  userForm;
  title: string;
  roles = ["Администратор", "Менеджер", "Исполнитель"];
  isNewUser: boolean = false;
  isEditUser: boolean = false;
  msg: string = "";
  @ViewChild('password') password: ElementRef;

  constructor(private dataService: DataService, private router: Router, private activatedRoute: ActivatedRoute, private fb: FormBuilder, private authService: AuthService, private _notifService: NotificationsService) { }

  ngOnInit() {
    this.userForm = this.fb.group({
      firstName: null,
      lastName: null,
      role: null,
      email: null,
      id: null,
      phone: null,
      imgLink: null,
      dateBirthday: null,
      notifications: null,
      username: null,
      latestLoginDate: null,
      notes: null
    });

    this.activatedRoute.params.forEach((params) => {
      if (params["userId"]) {
        this.getCurrentUser(params["userId"]);
      }

      else {
        this.isNewUser = true;
        this.isEditUser = true;
        this.msg = "Создать профиль"
      }
    })
  }

  editUser() {
    this.isEditUser = true;
    this.msg = "Редактировать профиль"
  }

  saveUser() {
    if (this.isNewUser) {
      this.userForm.value["password"] = this.password.nativeElement.value;

      if (!this.userForm.value.imgLink) this.userForm.value.imgLink = './../../assets/avatar.jpg';

      this.createUser();
    }
    else {
      this.updateUser();
    }
  }

  deleteUser() {
    this.dataService.deleteUser(this.userForm.value).subscribe(
      res => this.router.navigate(['/users'])
    );
  }

  getCurrentUser(id) {
    this.dataService.getUser(id).subscribe(
      res => {
        this.userForm.patchValue(res);
        console.log("Ответ с сервера:");
        console.log(res);

        console.log(this.userForm.getRawValue());
      }
    )
  }

  saveNotes(note) {
    this.userForm.value.notes = note.value;
    this.updateUser();
  }

  updateUser() {
    this.dataService.updateUser(this.userForm.value).subscribe(
      res => this._notifService.setNotification('Успешно сохранено', 1),
      err => this._notifService.setNotification('Ошибка сохранения', 2)
    );

    this.isEditUser = false;
  }

  createUser() {
    this.dataService.createUser(this.userForm.value).subscribe(
      res => this.router.navigate(['/users'])
    );
  }

}
