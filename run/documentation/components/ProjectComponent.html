import { Component, OnInit } from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import { DataService } from '../SHARED/services/data.service';
import { FormBuilder, Validators } from '@angular/forms';
import { AuthService } from '../SHARED/services/auth.service';
import { NgbDatepickerConfig, NgbDateStruct, NgbModal } from '@ng-bootstrap/ng-bootstrap';

/**
* Компонент проекта. Создание нового проекта либо отображение существующего.
*/
@Component({
  selector: 'app-project',
  templateUrl: './project.component.html',
  styleUrls: ['./project.component.css']
})
export class ProjectComponent implements OnInit {
  project; //Текущий обьъект проекта получанный через идентификатор
  projectForm; //Форма всего проекта связанная с шаблоном
  users;
  managerId: string;
  newProject: boolean = false;

  title: string; //Заголовок о создании нового проекта
  statuses = ["В работе", "Завершен", "Отменен", "Не определен"]

  constructor(private router: Router, private dataService: DataService,
    private fb: FormBuilder, private activatedRoute: ActivatedRoute,
    private authService: AuthService, config: NgbDatepickerConfig, private modalService: NgbModal) {


    // customize default values of datepickers used by this component tree
    config.minDate = { year: 1900, month: 1, day: 1 };
    config.maxDate = { year: 2099, month: 12, day: 31 };

    // days that don't belong to current month are not visible
    config.outsideDays = 'hidden';

    // weekends are disabled
    config.markDisabled = (date: NgbDateStruct) => {
      const d = new Date(date.year, date.month - 1, date.day);
      return d.getDay() === 0 || d.getDay() === 6;
    };
  }

  ngOnInit() {
    let id = this.activatedRoute.snapshot.params["id"];


    //Создаем форму проекта
    this.projectForm = this.fb.group({
      name: [{ value: '', disabled: this.authService.isWorker }, [Validators.required]],
      desc: [{ value: '', disabled: this.authService.isWorker }, [Validators.required]],
      status: [{ value: '', disabled: this.authService.isWorker }],
      author: [{ value: '', disabled: this.authService.isWorker }],
      id: [],
      taskQuantity: [],
      dateCreation: [],
      dateCompletion: [],
      managerId: '',
      manager: ''
    });

    //Определяем открыть проект или создать новый
    if (id) {
      this.dataService.getProject(id).subscribe((res) => {
        this.projectForm.patchValue(res);
        this.managerId = res["managerId"];
      });
    }

    else {
      this.title = 'Создать новый проект:';
      this.newProject = true;
    }

  }

  goBack() {
    this.router.navigate(['projects']);
  }

  saveProject() { // Создание нового проекта
    let these = this;
    if (!this.projectForm.value.id) {
      this.projectForm.value.author = this.authService.currentUser["firstName"] + " " + this.authService.currentUser["lastName"];
      this.projectForm.value["dateCreation"] = new Date();

      this.dataService.createProject(this.projectForm.value).subscribe((res) => {
        console.log(res);
        this.dataService.sendEvent(this.projectForm.value.author, this.projectForm.value.name, undefined, 'создал проект').subscribe();
        this.goBack();
      })
    }


    else { // Обновление проекта
      this.dataService.updateProject(this.projectForm.value).subscribe((result) => {

        // console.log(result);

        if (this.managerId !== result.managerId) {
          these.dataService.getTasks().subscribe(
            (res) => {

              console.log(res);
              for (let i in res) {
                console.log(res[i].managerId == this.managerId && res[i].projectId == result.id);
                if ( res[i].managerId == this.managerId && res[i].projectId == result.id) {
                  let task = res[i];
                  task.managerId = this.projectForm.value.managerId;
                  console.log(task);
                  these.dataService.updateTask(task).subscribe();
                }
              }
            }
          )
        }

        this.goBack();
      })
    }


  }


  //Вслыпающее окно участников
  open(content) {
    this.getUsers();
    this.modalService.open(content).result.then((result) => {
      this.projectForm.value.managerId = result["id"];
      this.projectForm.value.manager = result["lastName"] + ' ' + result["firstName"];
    }, (reason) => { });
  }

  getUsers() {
    this.dataService.getUsers().subscribe(
      res => {
        let users = [];

        for (let i in res) {
          if (res[i]["role"] == "Менеджер"){
            users.push(res[i]);
          }
        }
        this.users = users;
      }
    );
  }
 

}
