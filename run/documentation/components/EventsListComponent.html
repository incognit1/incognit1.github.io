import { Component, OnInit } from '@angular/core';
import { DataService } from '../SHARED/services/data.service';
import { HttpClient } from '@angular/common/http';
import { AuthService } from '../SHARED/services/auth.service';
import { PagerService } from '../SHARED/services/pager.service';

/**
* Компонент списка событий.
* Доступен только для администратора.
*/
@Component({
  selector: 'app-events-list',
  templateUrl: './events-list.component.html',
  styleUrls: ['./events-list.component.css']
})
export class EventsListComponent implements OnInit {
  events;
  // pager object
  pager: any = {};

  // paged items
  pagedItems: any[];

  constructor(private dataService: DataService, private pagerService: PagerService) { }

  ngOnInit() {
    this.dataService.getEvents().subscribe(
      (res) => {
        let stringArr: string[] = [];
        for (let i in res) {
          if (res[i]["taskName"] && res[i]["projectName"]) { // task
            let string = res[i]['user'] + ' ' + res[i]['action'] + " " + res[i]['taskName'] + ' в проекте ' + res[i]['projectName']
            stringArr.push(string);
          }

          else if (res[i]["taskName"] && !res[i]["projectName"]) { // comment
            let string = res[i]["user"] + " " + res[i]["action"] + " " + res[i]["taskName"]
            stringArr.push(string);
          }

          else if (!res[i]["taskName"] && res[i]["projectName"]) { // project
            let string = res[i]["user"] + " " + res[i]["action"] + " " + res[i]["projectName"]
            stringArr.push(string);
          }
        }

        this.events = stringArr.reverse();
        this.setPage(1);
      }
    )
  }

  setPage(page: number) {
    if (page < 1 || page > this.pager.totalPages) {
      return;
    }

    // get pager object from service
    this.pager = this.pagerService.getPager(this.events.length, page);

    console.log(this.pager);
    // get current page of items
    this.pagedItems = this.events.slice(this.pager.startIndex, this.pager.endIndex + 1);
  }


}
