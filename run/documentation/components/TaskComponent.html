import { Component, OnInit, Injectable } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { FormBuilder, Validators } from '@angular/forms';
import { Input } from '@angular/core';
import { AuthService } from '../SHARED/services/auth.service';
import { DataService } from '../SHARED/services/data.service';
import { NgbModal, ModalDismissReasons, NgbDateStruct, NgbCalendar, NgbDatepickerI18n } from '@ng-bootstrap/ng-bootstrap';
import { CustomDatepicker, I18n, DateOperations } from '../SHARED/services/calendar.service'
import { Task } from '../SHARED/models/task';

const now = new Date();
/**
* Компонент отображения, редактирования и создания задач.
*/
@Component({
  selector: 'app-task',
  templateUrl: './task.component.html',
  styleUrls: ['./task.component.css'],
  providers: [I18n, { provide: NgbDatepickerI18n, useClass: CustomDatepicker }, DateOperations]
})
export class TaskComponent implements OnInit {
  taskForm;
  title: string;
  statuses = ["В работе", "Выполнена", "Отмененная"];
  users;
  newTask: boolean = false;

  model: NgbDateStruct = { year: now.getFullYear(), month: now.getMonth() + 1, day: now.getDate() };
  disabled = true;

  isWorker = this.authService.isWorker;

  constructor(private activatedRoute: ActivatedRoute, private fb: FormBuilder,
    private authService: AuthService, private dataService: DataService, private router: Router, private modalService: NgbModal, calendar: NgbCalendar, private dateOperations: DateOperations) {

    //this.toDate = calendar.getNext(calendar.getToday(), 'd', 12);
  }

  ngOnInit() {
    this.taskForm = this.fb.group({
      name: [{ value: '', disabled: this.authService.isWorker }, [Validators.required]],
      desc: [{ value: '', disabled: this.authService.isWorker }, [Validators.required]],
      projectId: [],
      status: [{ disabled: this.authService.isWorker }, [Validators.required]],
      worker: [],
      workerId: [],
      author: [],
      authorId: '',
      dateCreation: [],
      dateStart: [],
      dateCompletion: [],
      projectName: [],
      managerId: [],
      id: []
    });

    let id = this.activatedRoute.snapshot.params['taskId'];

    //Загрузка данных из сервера если это не новая задача
    if (id && id !== 'new-task') {
      this.dataService.getTask(id).subscribe(
        res => {
          this.taskForm.patchValue(res);
          this.taskForm.value.dateCreation = new Date(res["dateCreation"]);

          if (!res["dateStart"]) {
            this.fromDate = this.dateOperations.fromModel(new Date());
          }

          else {
            this.taskForm.value.dateStart = new Date(res["dateStart"]);
            this.taskForm.value.dateCompletion = new Date(res["dateCompletion"]);

            this.fromDate = this.dateOperations.fromModel(new Date(res["dateStart"]));
            this.toDate = this.dateOperations.fromModel(new Date(res["dateCompletion"]));
          }

        },
        err => this.router.navigate(['../'], { relativeTo: this.activatedRoute })
      )
    }

    else if (id == 'new-task') {
      this.title = 'Добавить задачу';
      this.fromDate = this.dateOperations.fromModel(new Date());
      this.newTask = true;
    }



  }

  saveTask() {

    if (this.dateOperations.toModel(this.toDate) !== this.taskForm.value.dateCompletion) {
      this.taskForm.value.dateStart = this.dateOperations.toModel(this.fromDate);
      this.taskForm.value.dateCompletion = this.dateOperations.toModel(this.toDate);
    }

    //Обновление задачи
    if (this.taskForm.value.id) {
      if (this.toDate) {
        this.taskForm.value.dateStart = this.dateOperations.toModel(this.fromDate);
        this.taskForm.value.dateCompletion = this.dateOperations.toModel(this.toDate);
      }
      this.dataService.updateTask(this.taskForm.value).subscribe(
        (res) => {
          console.log('Данные сохранены');
          this.router.navigate(['../'], { relativeTo: this.activatedRoute });
        }
      );
    }

    //Создание задачи
    else {
      this.taskForm.value.projectId = this.activatedRoute.snapshot.params['id'];
      this.taskForm.value.author = this.authService.currentUser["lastName"] + ' ' + this.authService.currentUser["firstName"];
      this.taskForm.value.authorId = this.authService.currentUser["id"];

      this.taskForm.value.dateCreation = new Date();

      if (this.taskForm.value.worker) {
        this.taskForm.value.status = "Назначенная";
      } else {
        this.taskForm.value.status = "Новая";
      }

      if (this.toDate) {
        this.taskForm.value.dateStart = this.dateOperations.toModel(this.fromDate);
        this.taskForm.value.dateCompletion = this.dateOperations.toModel(this.toDate);
      }

      this.dataService.getProject(this.activatedRoute.snapshot.params['id']).subscribe(
        res => {
          this.taskForm.value.projectName = res["name"];
          this.taskForm.value.managerId = res["managerId"];
          this.dataService.createTask(this.taskForm.value).subscribe(
            task => {
              
              this.dataService.getProject(this.taskForm.value.projectId).subscribe(
                resp => {
                  let p = resp;
                  if (!p['taskQuantity']) p['taskQuantity'] = 0;
                  p['taskQuantity']++;
                  this.dataService.updateProject(p).subscribe();
                }
              )

              if (this.taskForm.value.worker) {
                this.dataService.getUser(this.taskForm.value.workerId).subscribe(
                  (response) => {
                    let user = response;
                    let notification = {
                      author: this.taskForm.value.author,
                      action: 'добавил Вас исполнителем к задаче',
                      object: {
                        type: 2,
                        name: task["name"],
                        id: task["id"]
                      },
                      wathed: false,
                      button: true
                    }

                    if (!user.notifications) user.notifications = [];

                    user.notifications.push(notification);

                    this.dataService.updateUser(user).subscribe();
                  }
                )
              }

              this.router.navigate(['../'], { relativeTo: this.activatedRoute })
            },
            (err) => console.log("Ошибка при создании задачи.")
          );
        }
      )
    }

  }


  //Вслыпающее окно работников
  open(content) {
    this.getUsers();
    this.modalService.open(content).result.then((result) => {
      this.taskForm.value.workerId = result["id"];
      this.taskForm.value.worker = result["lastName"] + ' ' + result["firstName"];
    }, (reason) => { });
  }


  getUsers() {
    this.dataService.getUsers().subscribe(
      res => {
        console.log(res);
        let users = [];
        for (let i in res) {
          if (res[i].role == "Исполнитель") {
            users.push(res[i]);
          }
        }

        this.users = users;
      }
    );
  }


  private getDismissReason(reason: any): string {
    if (reason === ModalDismissReasons.ESC) {
      return 'by pressing ESC';
    } else if (reason === ModalDismissReasons.BACKDROP_CLICK) {
      return 'by clicking on a backdrop';
    } else {
      return reason;
    }
  }



  // Календарь

  hoveredDate: NgbDateStruct;
  fromDate: NgbDateStruct;
  toDate: NgbDateStruct;


  onDateSelection(date: NgbDateStruct) {
    if (!this.fromDate && !this.toDate) {
      this.fromDate = date;
    } else if (this.fromDate && !this.toDate && after(date, this.fromDate)) {
      this.toDate = date;
    } else {
      this.toDate = null;
      this.fromDate = date;
    }

  }


  isHovered = date => this.fromDate && !this.toDate && this.hoveredDate && after(date, this.fromDate) && before(date, this.hoveredDate);
  isInside = date => after(date, this.fromDate) && before(date, this.toDate);
  isFrom = date => equals(date, this.fromDate);
  isTo = date => equals(date, this.toDate);

}


const equals = (one: NgbDateStruct, two: NgbDateStruct) => (one && two && two.year === one.year && two.month === one.month && two.day === one.day);


const before = (one: NgbDateStruct, two: NgbDateStruct) =>
  !one || !two ? false : one.year === two.year ? one.month === two.month ? one.day === two.day
    ? false : one.day < two.day : one.month < two.month : one.year < two.year;

const after = (one: NgbDateStruct, two: NgbDateStruct) =>
  !one || !two ? false : one.year === two.year ? one.month === two.month ? one.day === two.day
    ? false : one.day > two.day : one.month > two.month : one.year > two.year;

        <div *ngIf="!this.newTask">
          <p class="desc_title">
            <i class="fas fa-project-diagram"></i> Название проекта:</p> {{taskForm.value.projectName}}
        </div>

      </div>
      <p class="desc_title">
        <i class="fas fa-align-left"></i> Описание: </p>

      <textarea placeholder="Описание отсутствует" formControlName="desc"></textarea>

    <div class="calendar">
      <p class="desc_title">
        <i class="far fa-calendar"></i> Дата завершения: </p>

      <div *ngIf="this.authService.isWorker">
        <p>{{this.taskForm.value.dateCompletion | date : "dd/MM/yyyy"}}</p>
      </div>

      <ngb-datepicker *ngIf="!this.authService.isWorker" #dp (select)="onDateSelection($event)" [dayTemplate]="t">
      </ngb-datepicker>


      <ng-template #t let-date="date" let-focused="focused">

        <span class="custom-day" [class.focused]="focused" [class.range]="isFrom(date) || isTo(date) || isInside(date) || isHovered(date)"
          [class.faded]="isHovered(date) || isInside(date)" (mouseenter)="hoveredDate = date" (mouseleave)="hoveredDate = null">
          {{ date.day }}
        </span>
      </ng-template>

      <br>
      <br>
      <div *ngIf="!this.newTask">
        <p class="desc_title">Дата создания: </p>{{this.taskForm.value.dateCreation | date: 'short' }}
      </div>

    </div>

  </div>

  <div class="mb-center">
    <button routerLink="../" class="btn btn-secondary">Назад</button>
    <button (click)="saveTask()" class="btn btn-primary">Сохранить</button>
  </div>
</form>
