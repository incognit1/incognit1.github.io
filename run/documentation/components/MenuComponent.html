import { Component, OnInit } from '@angular/core';
import { AuthService } from '../SHARED/services/auth.service';
import { Router } from '@angular/router';
import { DataService } from '../SHARED/services/data.service';


/**
* Компонент меню. Используется во всех компонентах кроме модуля.
*/
@Component({
  selector: 'app-menu',
  templateUrl: './menu.component.html',
  styleUrls: ['./menu.component.css']
})
export class MenuComponent implements OnInit {
  events;
  currentUser;
  fixedProject;
  notifCounter: number = 0;
  notifications;


  constructor(private authService: AuthService, private router: Router, private dataService: DataService) { }

  ngOnInit() {
    this.currentUser = this.authService.currentUser;

    if (this.authService.currentUser["notifications"])

    for (let i in this.currentUser.notifications) {
      if (this.currentUser.notifications[i]["wathed"]) continue;
      else this.notifCounter++;
    }

  }

  logout() {
    this.authService.login = false;
    sessionStorage.removeItem('id');

    this.router.navigate(['login']);
  }

  getFixedProject() {
    this.dataService.getFixedProjects(this.currentUser).subscribe(
      res => {
        this.fixedProject = res;
      }
    );

  }

  getEvents() {
    this.dataService.getEvents().subscribe(
      (res) => {
        let stringArr: string[] = [];
        for (let i in res) {
          if (res[i]["taskName"] && res[i]["projectName"]) { // task
            let string = res[i]['user'] + ' ' + res[i]['action'] + " " + res[i]['taskName'] + ' в проекте ' + res[i]['projectName']
            stringArr.push(string);
          }

          else if (res[i]["taskName"] && !res[i]["projectName"]) { // comment
            let string = res[i]["user"] + " " + res[i]["action"] + " " + res[i]["taskName"]
            stringArr.push(string);
          }

          else if (!res[i]["taskName"] && res[i]["projectName"]) { // project
            let string = res[i]["user"] + " " + res[i]["action"] + " " + res[i]["projectName"]
            stringArr.push(string);
          }
        }

        this.events = stringArr.reverse().slice(0, 3);
      }
    )
  }

  wathNotif() {
    this.authService.getCurrentUser().subscribe(
      res => {
        if (res["notifications"] !== null) {
          let user = res;
          let length;
          (res["notifications"].length < 4) ? length = res["notifications"].length : length = 4;
          this.notifications = res["notifications"].reverse().slice(0, length);
          for (let i in this.notifications) {
            this.notifications[i]["wathed"] = true;
          }

          user["notifications"] = this.notifications.slice();

          this.dataService.updateUser(user).subscribe(
            res => this.notifCounter = 0
          );
        }
      }
    )






  }

  accept(id, type) {
    if (type == 2) {
      this.dataService.getTask(id).subscribe(
        (res) => {
          let task = res;

          task["status"] = "В работе";
          this.dataService.updateTask(task).subscribe();

          let arr = this.currentUser.notifications;

          this.dataService.getUser(this.currentUser.id).subscribe(
            res => {
              let user = res;
              console.log(res)
              for (let i in user.notifications) {
                if (user.notifications[i]['object'].id == id && user.notifications[i].button) {
                  console.log('hello');
                  user.notifications[i].button = false;
                }
              }

              this.getEvents();
              this.dataService.updateUser(user).subscribe();
            }
          )





        }
      )
    }
  }

}
