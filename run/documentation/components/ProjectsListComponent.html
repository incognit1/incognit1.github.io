import { Component, OnInit } from '@angular/core';
import { DataService } from '../SHARED/services/data.service';
import { Router } from '@angular/router';
import { AuthService } from '../SHARED/services/auth.service';
import { NgbModal, ModalDismissReasons } from '@ng-bootstrap/ng-bootstrap';
import { Project } from '../SHARED/models/projects';
import { NgbDropdownConfig } from '@ng-bootstrap/ng-bootstrap';
import { AppearanceTrigger } from '../SHARED/animations';

/**
* Компонент списка проектов.
*/
@Component({
  selector: 'app-projects-list',
  templateUrl: './projects-list.component.html',
  styleUrls: ['./projects-list.component.css'],
  providers: [NgbDropdownConfig],
  animations: [AppearanceTrigger]
})
export class ProjectsListComponent implements OnInit {
  projects: Project[] = [];
  checkedProject;
  workers = [];

  errorMessage: string;
  constructor(private dataService: DataService, private router: Router,
    private authService: AuthService, private modalService: NgbModal, config: NgbDropdownConfig) {
    config.placement = 'bottom-right';
  }

  ngOnInit() {
    this.getProjects();
  }

  private getProjects() {
    this.dataService.getProjects().subscribe(
      projects => {
        this.projects = projects;
        
      },
      error => this.errorMessage = error
    );
  }

  deleteProject(project) {
    this.dataService.deleteProject(project).subscribe((res) => {
      this.getProjects();
    })
  }

  bindProject(project) {
    this.dataService.bindProject(project, this.authService.currentUser)
  }

  //Вслыпающее окно участников
  open(content, project) {
    console.log(project.id);
    this.getUsersInProject(project.id);

    this.checkedProject = project;
    this.modalService.open(content).result.then((result) => {
      this.workers = [];
      console.log(`Closed with: ${result}`);
    }, (reason) => { this.workers = [] });
  }

  getUsersInProject(id) {
    this.dataService.getTasks(id).subscribe(
      res => {
        nextTask: for (let i in res) {
          if (!res[i]["workerId"]) {
            continue nextTask;
          }

          this.dataService.getUser(res[i]["workerId"]).subscribe(
            result => {

              for (let j in this.workers) { // Проверка на повторение одного пользователя в разных задачах
                if (res[i]["workerId"] == this.workers[j]["workerId"]) return;
              }

              this.workers.push(
                {
                  worker: result["firstName"] + " " + result["lastName"],
                  workerId: res[i]["workerId"],
                  img: result["imgLink"]
                }
              );
            }
          )



        }


      }
    );
  }

  private getDismissReason(reason: any): string {
    if (reason === ModalDismissReasons.ESC) {
      return 'by pressing ESC';
    } else if (reason === ModalDismissReasons.BACKDROP_CLICK) {
      return 'by clicking on a backdrop';
    } else {
      return `with: ${reason}`;
    }
  }

}
